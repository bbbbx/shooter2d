cmake_minimum_required(VERSION 3.0)

project(test-sdl2)

find_package(SDL2 REQUIRED)
if(SDL2_FOUND)
    include_directories(${SDL2_INCLUDE_DIRS}/../)
    add_executable(
        main
        src/main.cpp
        src/init.c
        src/input.c
        src/draw.c
        src/stage.c
        src/utils.c
        src/sound.c
        src/text.c
        src/highscores.c
    )
    target_link_libraries(main ${SDL2_LIBRARIES})
endif()

    /opt/homebrew/include
)
